import numpy as np
import scipy.interpolate as interp
from datetime import datetime,timedelta
date_range = pandas.date_range(start="2021-07-20 00:00:00",end="2021-07-23 12:00:00",freq="6H")
print(date_range)

#內插風速，先計算風速
for tau in range(0,85,6):
    valdtime = date_range[0] + timedelta(hours=tau)
    valdstr = valdtime.strftime("%Y%m%d%H00")
    print(valdstr)
    if tau < 10:
        model = pygrib.open("./21072000/M-A0064-21072000-00{:d}.grb2".format(tau))
    elif tau > 99:
        model = pygrib.open("./21072000/M-A0064-21072000-{:d}.grb2".format(tau))
    else:
        model = pygrib.open("./21072000/M-A0064-21072000-0{:d}.grb2".format(tau))
    model_u10 = model.select()[:][66]["values"] 
    model_v10 = model.select()[:][67]["values"]
    wspd10 = np.sqrt(model_u10**2 + model_v10**2)
    obs =  pandas.read_csv("./21072000/obs/{}".format(valdstr))
    print(len(obs["lon"]),len(obs["lat"]))
    interpdata = interp.griddata((lons.flatten(),lats.flatten()),wspd10.flatten(),(obs["lon"],obs["lat"]),method='nearest')
    obs["WSPD_inter"] = interpdata
    obs.to_csv("./21072000/obs_interp/utf8_{}_interp.csv".format(valdstr),index=False,encoding='utf-8-sig')

import matplotlib.pyplot as plt
obspack =[]
modelpack = []
ticks = []
for tau in range(0,85,6):
    valdtime = date_range[0] + timedelta(hours=tau)
    valdstr = valdtime.strftime("%Y%m%d%H00")
    ticks.append(valdstr[2:-2])
    obs_interp =  pandas.read_csv("./21072000/obs_interp/{}_interp.csv".format(valdstr))
    stnid = 466920
    stnobs = obs_interp.query("stnid == '{:d}'".format(stnid))
    obspack.append(stnobs["WDSD"])
    modelpack.append(stnobs["WSPD_inter"])
print(ticks)
fig = plt.figure(figsize=(16,8))
plt.plot(obspack,"r-",label="OBS Wind speed")
plt.plot(modelpack,"b-",label="Model Wind speed")
plt.xticks(list(range(len(ticks))),ticks,rotation=30)
plt.xlabel("Time",fontsize=16)
plt.ylabel("Wind speed (m/s)",fontsize=16)
plt.legend()
plt.show()
